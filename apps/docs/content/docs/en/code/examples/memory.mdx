# Memory

```ts
import { FileMemoryStore } from '@a3s-lab/code';

const session = agent.session('/repo', {
  memoryStore: new FileMemoryStore('./.a3s/memory'),
});

await session.rememberSuccess('docs rewrite', ['grep', 'read'], 'stale HCL references removed');
const memories = await session.recallSimilar('docs release rewrite', 5);
```
