# Agent Harness Protocol Example

This example shows how to attach an external harness server to an A3S Code session using the [Agent Harness Protocol](/docs/en/code/harness). The harness server blocks dangerous Bash commands and logs sensitive output detections — all from a standalone Python process.

## Harness server (`harness.py`)

Save this as `harness.py` alongside your agent config. It reads events from stdin and writes responses to stdout.

```python
#!/usr/bin/env python3
"""
Minimal AHP harness server — blocks dangerous Bash commands,
logs sensitive output detections to stderr.
"""
import json, re, sys
from typing import Any

BLOCKED_PATTERNS = [
    (r"rm\s+-[rf]{1,2}\s+/", "recursive delete from root"),
    (r"\bdd\s+if=",           "raw disk write"),
    (r"curl\s+.*\|\s*(ba)?sh","pipe curl to shell"),
    (r":\(\)\s*\{.*:\|:&\s*\}","fork bomb"),
]

SENSITIVE_PATTERNS = [
    r"(?i)(password|secret|token|api[_-]?key)\s*[:=]\s*\S+",
    r"[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}",
]

def log(msg: str) -> None:
    print(f"[AHP] {msg}", file=sys.stderr, flush=True)

def on_pre_tool_use(payload: dict, depth: int) -> dict:
    tool    = payload.get("tool", "")
    command = payload.get("args", {}).get("command", "") if isinstance(payload.get("args"), dict) else ""
    sid     = payload.get("session_id", "?")
    tag     = f"d{depth}" if depth else "root"

    if tool == "Bash" and command:
        for pattern, desc in BLOCKED_PATTERNS:
            if re.search(pattern, command, re.IGNORECASE):
                log(f"BLOCK [{sid}][{tag}] {desc}: {command[:80]}")
                return {"action": "block", "reason": f"AHP: {desc}"}

    log(f"ALLOW [{sid}][{tag}] {tool}: {command[:60] or '(no command)'}")
    return {"action": "continue"}

def on_post_tool_use(payload: dict, depth: int) -> None:
    output = (payload.get("result") or {}).get("output", "")
    sid    = payload.get("session_id", "?")
    tag    = f"d{depth}" if depth else "root"
    if output:
        for p in SENSITIVE_PATTERNS:
            if re.search(p, output):
                log(f"ALERT [{sid}][{tag}] sensitive data in {payload.get('tool','?')} output")
                break

BLOCKING      = {"pre_tool_use": on_pre_tool_use, "pre_prompt": lambda p, d: {"action": "continue"}}
NOTIFICATIONS = {"post_tool_use": on_post_tool_use}

log("AHP server started")
for raw in sys.stdin:
    line = raw.strip()
    if not line:
        continue
    try:
        msg = json.loads(line)
    except json.JSONDecodeError as e:
        log(f"ERROR parse: {e}")
        continue

    params     = msg.get("params", {})
    event_type = params.get("event_type", "")
    payload    = params.get("payload", {})
    depth      = params.get("meta", {}).get("depth", 0)
    req_id     = msg.get("id")

    if req_id is None:
        if event_type in NOTIFICATIONS:
            NOTIFICATIONS[event_type](payload, depth)
        continue

    result = BLOCKING.get(event_type, lambda p, d: {"action": "continue"})(payload, depth)
    print(json.dumps({"jsonrpc": "2.0", "id": req_id, "result": result}), flush=True)

log("AHP server exiting")
```

## Attaching the harness

<Tabs items={['Rust', 'TypeScript', 'Python']}>
<Tab value="TypeScript">

```typescript
import { Agent, HarnessServer } from '@a3s-lab/code';

const agent = await Agent.create('agent.hcl');

const session = agent.session('.', {
  builtinSkills: true,
  harnessServer: new HarnessServer('python3', ['harness.py']),
});

// The harness intercepts every tool call — including inside sub-agents
const result = await session.send('List all TypeScript files and count lines of code');
console.log(result.text);

// To test blocking:
// const r = await session.send('Run: rm -rf /tmp/test');
// The agent will report that the Bash tool was blocked.
```

```python
from a3s_code import Agent, HarnessServer, SessionOptions

agent = Agent.create('agent.hcl')

opts = SessionOptions()
opts.harness_server = HarnessServer('python3', ['harness.py'])
opts.builtin_skills = True

session = agent.session('.', opts)

# The harness intercepts every tool call — including inside sub-agents
result = session.send('List all Python files and count lines of code')
print(result.text)

# To test blocking:
```

## Multi-agent scenario

When the agent uses the `task` tool to spawn sub-agents, the harness server automatically receives events from all sub-agents. The `meta.depth` field increments at each level:

```python
# harness.py output (stderr) for a multi-agent run:
# [AHP] START [root-session][root] model=anthropic/claude-...
# [AHP] ALLOW [root-session][root] task: (no command)
# [AHP] START [child-session][d1]  model=anthropic/claude-...
# [AHP] ALLOW [child-session][d1]  Bash: ls -la
# [AHP] ALLOW [child-session][d1]  Glob: **/*.rs
# [AHP] START [grandchild][d2]     model=anthropic/claude-...
# [AHP] BLOCK [grandchild][d2]     recursive delete from root: rm -rf /
```

<Callout type="info">
  The harness server receives a single multiplexed stream of events from all agents. Use `session_id` to correlate events to specific sub-agents, and `depth` to apply depth-aware policies.
</Callout>

## Running the harness standalone (testing)

You can test the harness server independently without A3S Code:

```bash
# Test: allow a safe command
echo '{"jsonrpc":"2.0","id":1,"method":"harness/event","params":{"event_type":"pre_tool_use","payload":{"session_id":"test","tool":"Bash","args":{"command":"ls -la"},"working_directory":"/","recent_tools":[]},"meta":{"depth":0}}}' \
  | python3 harness.py
# → {"jsonrpc": "2.0", "id": 1, "result": {"action": "continue"}}

# Test: block a dangerous command
echo '{"jsonrpc":"2.0","id":2,"method":"harness/event","params":{"event_type":"pre_tool_use","payload":{"session_id":"test","tool":"Bash","args":{"command":"rm -rf /"},"working_directory":"/","recent_tools":[]},"meta":{"depth":1}}}' \
  | python3 harness.py
# → {"jsonrpc": "2.0", "id": 2, "result": {"action": "block", "reason": "AHP: recursive delete from root"}}
```
