# Middleware

A3S Code provides an Express-style middleware system that allows you to intercept and modify context during session creation. Middleware supports both synchronous and asynchronous execution.

## Architecture

```
Agent.session(workspace, options)
  ↓
MiddlewarePipeline.execute(ctx)
  ├─ Middleware 1 (sync/async)
  ├─ Middleware 2 (sync/async)
  └─ Middleware N (sync/async)
  ↓
AgentSession created
```

## Basic Usage

### Synchronous Middleware

<Tabs groupId="lang" items={['TypeScript', 'Python']}>
<Tab value="TypeScript">
```typescript
import { MiddlewarePipeline, MiddlewareContext } from '@a3s-lab/code';

const pipeline = new MiddlewarePipeline();

pipeline.useMiddleware((ctx: MiddlewareContext) => {
  console.log(`Session: ${ctx.sessionId}, Workspace: ${ctx.workspace}`);

  // Modify context
  ctx.prompt = `[Enhanced] ${ctx.prompt}`;

  return { resultType: 'continue' };
});

// Execute pipeline
const result = await pipeline.execute({
  sessionId: 'session-123',
  workspace: '/project',
  prompt: 'Refactor auth module',
});
```

### Asynchronous Middleware

TypeScript supports async middleware via the `AsyncMiddlewarePipeline` wrapper:

<Tabs groupId="lang" items={['TypeScript', 'Python']}>
<Tab value="TypeScript">
```typescript
import { MiddlewarePipeline, MiddlewareContext } from '@a3s-lab/code';

// AsyncMiddlewarePipeline wrapper
class AsyncMiddlewarePipeline {
  private pipeline: MiddlewarePipeline;
  private pendingPromises: Promise<any>[] = [];

  constructor() {
    this.pipeline = new MiddlewarePipeline();
  }

  use(middleware: (ctx: MiddlewareContext) => any | Promise<any>): this {
    const isAsync = middleware.constructor.name === 'AsyncFunction';

    if (isAsync) {
      const wrapped = (ctx: MiddlewareContext) => {
        const promise = middleware(ctx);
        this.pendingPromises.push(promise);
        return { resultType: 'continue' };
      };
      this.pipeline.useMiddleware(wrapped);
    } else {
      this.pipeline.useMiddleware(middleware);
    }

    return this;
  }

  async execute(ctx: MiddlewareContext): Promise<MiddlewareContext> {
    this.pendingPromises = [];
    const result = await this.pipeline.execute(ctx);

    // Wait for all async middleware to complete
    if (this.pendingPromises.length > 0) {
      const results = await Promise.all(this.pendingPromises);
      for (const middlewareResult of results) {
        if (middlewareResult.resultType === 'abort') {
          throw new Error(`Middleware aborted: ${middlewareResult.reason}`);
        }
      }
    }

    return result;
  }
}

// Use async middleware
const pipeline = new AsyncMiddlewarePipeline();

pipeline.use(async (ctx) => {
  console.log('⏳ Async middleware: starting...');
  await new Promise(resolve => setTimeout(resolve, 100));
  console.log('✅ Async middleware: completed');
  return { resultType: 'continue' };
});

await pipeline.execute(ctx);
```

## Middleware Context

### MiddlewareContext Fields

| Field | Type | Description |
|-------|------|-------------|
| `sessionId` | `string` | Unique session identifier |
| `workspace` | `string` | Workspace directory path |
| `prompt` | `string?` | Optional initial prompt |

### Middleware Result

| Result Type | Description |
|-------------|-------------|
| `continue` | Continue to next middleware |
| `abort` | Abort pipeline execution |

## Complete Example

<Tabs groupId="lang" items={['TypeScript', 'Python']}>
<Tab value="TypeScript">
```typescript
import { MiddlewarePipeline, MiddlewareContext } from '@a3s-lab/code';

const pipeline = new MiddlewarePipeline();

// 1. Logging
pipeline.useMiddleware((ctx) => {
  console.log(`🟢 Session ${ctx.sessionId} starting`);
  return { resultType: 'continue' };
});

// 2. Validation
pipeline.useMiddleware((ctx) => {
  if (!ctx.workspace) {
    return { resultType: 'abort', reason: 'Workspace required' };
  }
  return { resultType: 'continue' };
});

// 3. Enhancement
pipeline.useMiddleware((ctx) => {
  ctx.prompt = `[Enhanced] ${ctx.prompt || ''}`;
  return { resultType: 'continue' };
});

// Execute
const result = await pipeline.execute({
  sessionId: 'session-123',
  workspace: '/project',
  prompt: 'Refactor auth',
});

console.log('Pipeline completed:', result);
```
</Tab>

```python
from a3s_code import MiddlewarePipeline

pipeline = MiddlewarePipeline()

# 1. Logging
def logging_middleware(ctx):
    print(f"🟢 Session {ctx.session_id} starting")
    return {"result_type": "continue"}

# 2. Validation
def validation_middleware(ctx):
    if not ctx.workspace:
        return {"result_type": "abort", "reason": "Workspace required"}
    return {"result_type": "continue"}

# 3. Enhancement
def enhancement_middleware(ctx):
    ctx.prompt = f"[Enhanced] {ctx.prompt or ''}"
    return {"result_type": "continue"}

pipeline.use_middleware(logging_middleware)
pipeline.use_middleware(validation_middleware)
pipeline.use_middleware(enhancement_middleware)

# Execute
result = pipeline.execute({
    "session_id": "session-123",
    "workspace": "/project",
    "prompt": "Refactor auth",
})

print("Pipeline completed:", result)
```
