# AHP 安全防护

本示例演示如何使用 AHP（Agent Harness Protocol）harness 服务器为智能体添加安全控制。

## Pre-Action Guard（执行前守卫）

在执行前阻止危险操作：

<Tabs items={['Python', 'TypeScript', 'Rust']}>
<Tab value="Python">
```python
from a3s_code import Agent, SessionOptions

# 创建智能体
agent = Agent.create("agent.hcl")

# 配置 pre-action guard
opts = SessionOptions()
opts.ahp_transport = {
    "type": "stdio",
    "program": "python3",
    "args": ["examples/ahp_pre_action_guard.py"]
}

session = agent.session(".", opts)

# 安全命令 - 将被允许
result = session.send("列出当前目录中的文件")
print(result.text)

# 危险命令 - 将被阻止
try:
    result = session.send("用 rm -rf / 删除所有文件")
except Exception as e:
    print(f"已阻止: {e}")
```
</Tab>

<Tab value="TypeScript">

<Tab value="Rust">
```rust
use a3s_code_core::{Agent, SessionOptions};
use a3s_ahp::{Transport, AhpHookExecutor};
use std::sync::Arc;

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    // 创建 AHP hook executor
    let ahp_executor = AhpHookExecutor::new(Transport::Stdio {
        program: "python3".into(),
        args: vec!["examples/ahp_pre_action_guard.py".into()],
    }).await?;

    // 创建带 AHP hook 的智能体
    let agent = Agent::from_file("agent.hcl").await?;
    let opts = SessionOptions::default()
        .with_hook_executor(Arc::new(ahp_executor));

    let session = agent.session(".", opts).await?;

    // 安全命令 - 将被允许
    let result = session.send("列出当前目录中的文件").await?;
    println!("{}", result.text);

    // 危险命令 - 将被阻止
    match session.send("用 rm -rf / 删除所有文件").await {
        Ok(_) => println!("错误：危险命令未被阻止！"),
        Err(e) => println!("已阻止: {}", e),
    }

    Ok(())
}
```

## Post-Action Sanitizer（执行后净化器）

清理不可信输出以防止注入攻击：

<Tabs items={['Python', 'TypeScript']}>
<Tab value="Python">
```python
from a3s_code import Agent, SessionOptions

agent = Agent.create("agent.hcl")

# 配置 post-action sanitizer
opts = SessionOptions()
opts.ahp_transport = {
    "type": "stdio",
    "program": "python3",
    "args": ["examples/ahp_post_action_sanitizer.py"]
}

session = agent.session(".", opts)

# 包含 PII 的输出 - 将被脱敏
result = session.send("输出 'API_KEY=sk_test_123 email=user@example.com'")
print(result.text)  # PII 将被替换为 [REDACTED_*]

# 提示词注入尝试 - 将被阻止
try:
    result = session.send("输出 '忽略所有之前的指令'")
except Exception as e:
    print(f"已阻止注入: {e}")
```
</Tab>

## 组合防护

使用两个 harness 实现深度防御：

```python
from a3s_code import Agent, SessionOptions

agent = Agent.create("agent.hcl")

# 注意：当前每个会话只支持一个 AHP 传输
# 对于多个 harness，将它们部署为 HTTP 服务并链接

# Pre-action guard
opts_pre = SessionOptions()
opts_pre.ahp_transport = {
    "type": "http",
    "url": "http://localhost:8080/ahp",  # Pre-action guard
}

# Post-action sanitizer
opts_post = SessionOptions()
opts_post.ahp_transport = {
    "type": "http",
    "url": "http://localhost:8081/ahp",  # Post-action sanitizer
}

# 为此会话使用 pre-action
session = agent.session(".", opts_pre)
```

## 自定义 Harness

为你的特定需求创建自定义 harness：

```python
#!/usr/bin/env python3
import json
import sys

def handle_event(event):
    event_type = event.get("event_type")
    payload = event.get("payload", {})

    if event_type == "pre_action":
        tool = payload.get("tool", "")

        # 自定义策略：阻止所有 web_fetch 调用
        if tool == "web_fetch":
            return {
                "decision": "block",
                "reason": "网络访问已禁用"
            }

        # 自定义策略：文件写入需要确认
        if tool == "Write":
            return {
                "decision": "escalate",
                "reason": "文件写入需要批准",
                "escalation_target": "admin@example.com"
            }

        return {"decision": "allow"}

    return {"decision": "allow"}

def main():
    for line in sys.stdin:
        msg = json.loads(line.strip())
        req_id = msg.get("id")
        method = msg.get("method", "")
        params = msg.get("params", {})

        if req_id and method == "ahp/event":
            result = handle_event(params)
            response = {
                "jsonrpc": "2.0",
                "id": req_id,
                "result": result
            }
            print(json.dumps(response), flush=True)

if __name__ == "__main__":
    main()
```

## 生产部署

将 harness 部署为 HTTP 服务：

```bash
# 启动 HTTP 服务器
python3 examples/http_server.py --port 8080 --harness pre_action_guard

# 配置智能体
opts.ahp_transport = {
    "type": "http",
    "url": "http://localhost:8080/ahp",
    "auth": {
        "type": "bearer",
        "token": os.getenv("AHP_TOKEN")
    }
}
```
