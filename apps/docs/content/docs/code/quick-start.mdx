## Hello World

<Tabs groupId="lang" items={['Rust', 'Python', 'Node.js']}>
<Tab value="Rust">
```rust
use a3s_code_core::{Agent, AgentEvent};

#[tokio::main]
async fn main() -> anyhow::Result<()> {
    let agent = Agent::new("agent.hcl").await?;
    let session = agent.session(".", None)?;

    // Non-streaming
    let result = session.send("What does this project do?").await?;
    println!("{}", result.text);

    // Streaming
    let (mut rx, _handle) = session.stream("List all source files").await?;
    while let Some(event) = rx.recv().await {
        if let AgentEvent::TextDelta { text } = event {
            print!("{text}");
        }
    }

    // Direct tool calls
    let files = session.glob("**/*.rs").await?;
    println!("Found {} Rust files", files.len());

    let output = session.bash("cargo test --lib 2>&1 | tail -5").await?;
    println!("{output}");

    Ok(())
}
```
</Tab>
<Tab value="Python">
```python
from a3s_code import Agent

agent = Agent.create("agent.hcl")
session = agent.session(".")

# Non-streaming
result = session.send("What does this project do?")
print(result.text)

# Streaming
for event in session.stream("List all source files"):
    if event.event_type == "text_delta":
        print(event.text, end="", flush=True)

# Direct tool calls
files = session.glob("**/*.py")
print(f"\nFound {len(files)} Python files")

output = session.bash("python -m pytest --co -q 2>&1 | tail -5")
print(output)
```
</Tab>
<Tab value="Node.js">
```typescript
const { Agent } = require('@a3s-lab/code');

const agent = await Agent.create('agent.hcl');
const session = agent.session('.');

// Non-streaming
const result = await session.send('What does this project do?');
console.log(result.text);

// Streaming
const events = await session.stream('List all source files');
for (const event of events) {
  if (event.type === 'text_delta') process.stdout.write(event.text);
}

// Direct tool calls
const files = await session.glob('**/*.ts');
console.log(`\nFound ${files.length} TypeScript files`);

const output = await session.bash('npm test 2>&1 | tail -5');
console.log(output);
```

## Configuration

Create an `agent.hcl` (or `agent.json`) file. HCL is preferred.

```hcl
default_model = "anthropic/claude-sonnet-4-20250514"

providers {
  name    = "anthropic"
  api_key = "sk-ant-..."

  models {
    id        = "claude-sonnet-4-20250514"
    name      = "Claude Sonnet 4"
    tool_call = true
    cost {
      input       = 3.0
      output      = 15.0
      cache_read  = 0.3
      cache_write = 3.75
    }
    limit {
      context = 200000
      output  = 8192
    }
  }
}

providers {
  name    = "openai"
  api_key = "sk-..."

  models {
    id        = "gpt-4o"
    name      = "GPT-4o"
    tool_call = true
  }
}
```

You can also pass inline JSON or HCL strings directly to `Agent.create()` / `Agent::new()`.

See [Configuration](/docs/code/sessions#configuration) for the full reference.

## Per-Session Model Override

Override the default model for a specific session using `provider/model` format:

<Tabs groupId="lang" items={['Rust', 'Python', 'Node.js']}>
<Tab value="Rust">
```rust
use a3s_code_core::SessionOptions;

let session = agent.session("/my-project", Some(
    SessionOptions::new()
        .with_model("openai/gpt-4o")
))?;
```
</Tab>
<Tab value="Python">
```python
session = agent.session("/my-project", model="openai/gpt-4o")
```
</Tab>
<Tab value="Node.js">
```typescript
const session = agent.session('/my-project', {
  model: 'openai/gpt-4o',
});
```
</Tab>
